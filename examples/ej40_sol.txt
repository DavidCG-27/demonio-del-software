//PROTOTYPE
// Interfaz Prototype
interface Prototype {
    Prototype clone();
}

// Clase concreta Document
class Document implements Prototype {
    private String title;
    private String content;

    public Document(String title, String content) {
        this.title = title;
        this.content = content;
    }

    @Override
    public Prototype clone() {
        return new Document(this.title, this.content);
    }

    public void show() {
        System.out.println("Documento: " + title + "\nContenido: " + content);
    }
}

// Uso
public class Main {
    public static void main(String[] args) {
        Document doc1 = new Document("Reporte", "Este es el contenido...");
        Document doc2 = (Document) doc1.clone();

        doc2.show();
    }
}
