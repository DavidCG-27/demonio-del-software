// Preparador de té
class TeaMaker {
    public void prepare() {
        System.out.println("Hirviendo agua...");
        System.out.println("Añadiendo hojas de té...");
        System.out.println("Vertiendo en la taza...");
        System.out.println("Añadiendo limón...");
        System.out.println("Té preparado.");
    }
}
