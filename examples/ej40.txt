class Document {
    private String title;
    private String content;

    public Document(String title, String content) {
        this.title = title;
        this.content = content;
    }

    public Document duplicate() {
        return new Document(this.title, this.content);
    }

    public void show() {
        System.out.println("Documento: " + title + "\nContenido: " + content);
    }
}

// Uso
public class Main {
    public static void main(String[] args) {
        Document doc1 = new Document("Reporte", "Este es el contenido...");
        Document doc2 = doc1.duplicate();

        doc2.show();
    }
}
