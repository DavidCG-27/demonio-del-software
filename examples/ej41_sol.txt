//PROTOTYPE
// Interfaz Prototype
interface Prototype {
    Prototype clone();
}

// Clase concreta GameCharacter
class GameCharacter implements Prototype {
    public String name;
    public int health;
    public int attackPower;

    public GameCharacter(String name, int health, int attackPower) {
        this.name = name;
        this.health = health;
        this.attackPower = attackPower;
    }

    @Override
    public Prototype clone() {
        return new GameCharacter(this.name, this.health, this.attackPower);
    }
}
